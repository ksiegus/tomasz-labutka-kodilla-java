SELECT I.ID, I.SUMMARY, I.DESCRIPTION, IL.NAME AS LISTNAME
FROM ISSUES I JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID;


SELECT I.ID, I.SUMMARY, I.DESCRIPTION, IL.NAME AS LISTNAME, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID
	JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.ID, U.FIRSTNAME, U.LASTNAME, count(*) as ISSUES_COUNT
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID;
